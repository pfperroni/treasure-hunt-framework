294  void reset(Region<P> *r, Solution<P, pSize, F, fSize, V, vSize> *bias = NULL) {
295  if (r == NULL) throw std::invalid_argument("Region cannot be null.");
296  map<int, Dimension<P>*> *originalDimensions = r->getOriginalDimensions();
297  if (n != (int) originalDimensions->size()) {
298  throw std::invalid_argument("The number of dimensions does not match.");
299  }
300 
301  map<Dimension<P>*, Partition<P>*> *partitions = r->getPartitions();
302  Partition<P> *partition;
303  Position<P, pSize> *pos;
304  for (int i = 0; i < n; i++) {
305  // Use originalDimensions to map the correct sequence of dimensions on partitions.
306  partition = (*partitions)[(*originalDimensions)[i]];
307  if(bias != NULL) { // If a bias is provided.
308  pos = (*bias)[i];
309  if(THUtil::randUniformDouble(seed, 0, 1) < 0.5) { // For half of the dimensions:
310  // Reset close to the bias.
311  positions[i] = THUtil::randNormalDouble(seed, pos->getLowerValue()*0.99, pos->getUpperValue()*1.01);
312  } // For the other half:
313  else positions[i] = pos; // Reset exactly on the same position as the bias.
314  }
315  else { // If no bias is provides:
316  // Reset to a random position within original "anchor" sub-region.
317  positions[i] = THUtil::randUniformDouble(seed, partition->getStartPoint(), partition->getEndPoint());
318  }
319  // Make sure the new position is inside the boundaries.
320  positions[i].adjustUpperBound(partition->getEndPoint());
321  positions[i].adjustLowerBound(partition->getStartPoint());
322  }
323  }
