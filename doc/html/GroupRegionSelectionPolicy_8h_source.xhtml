42  Region<P>* internalLoop(Region<P> *region, vector<int> *hierarchy, t_node *node, int ID) {
43  if(node->getID() == ID){ // If the recursive iteration found the ID's sub-region.
44  return new Region<P>(region);
45  }
46  // Obtains the correct subdivision of node's search space by finding
47  // the node's child that is in the top of ID's parentage.
48  std::vector<t_node*> *children = node->getChildren();
49  for(int childPos=0, top=hierarchy->size()-1; childPos < (int)children->size(); childPos++) {
50  // If this child is in top of ID's parentage.
51  if((*children)[childPos]->getID() == (*hierarchy)[top]){
52  // Find child's coordinate in dimension grouping.
53  int coord[nGroups];
54  memset(coord, 0, sizeof(int)*nGroups);
55  for(int pos=childPos, g=nGroups-1, base; g >= 0; g--) {
56  base = pow(K, g);
57  if(base <= pos) {
58  coord[g] = pos / base; // Simplified coordinate in group g.
59  pos %= base;
60  }
61  }
62  // Convert: child's coordinate in dimension grouping -> search space boundaries inside parent's subregion.
63  int nDim = region->getNDimensions();
64  int dimPerGroup = nDim / nGroups;
65  Region<P> subRegion(region);
66  Partition<P>* partition;
67  P delta, minimum;
68  for(int d=0, g=0; d < nDim; d++){
69  partition = (*region)[d]; // Perform the partitioning using the sequential order of dimension's ID.
70  delta = (partition->getEndPoint() - partition->getStartPoint()) / K;
71  minimum = partition->getStartPoint() + coord[g] * delta;
72  subRegion[d]->setStartPoint(minimum);
73  subRegion[d]->setEndPoint((coord[g]<K-1) ? minimum + delta : partition->getEndPoint());
74  if((d+1) % dimPerGroup == 0) g++; // Move to the next group.
75  }
76  hierarchy->pop_back(); // Remove the node that is in the top of ID's parentage.
77  // Re-partition the just found sub-region according to ID's descendancy.
78  return internalLoop(&subRegion, hierarchy, (*children)[childPos], ID);
79  }
80  }

83  }
