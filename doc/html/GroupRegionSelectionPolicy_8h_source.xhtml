27  Region<P>* internalLoop(Region<P> *region, vector<int> *hierarchy, t_node *node, int ID) {
28  if(node->getID() == ID){ // If the recursive iteration found the ID's sub-region.
29  return new Region<P>(region);
30  }
31  // Obtains the correct subdivision of node's search space by finding
32  // the node's child that is in the top of ID's parentage.
33  std::vector<t_node*> *children = node->getChildren();
34  for(int childPos=0, top=hierarchy->size()-1; childPos < (int)children->size(); childPos++) {
35  // If this child is in top of ID's parentage.
36  if((*children)[childPos]->getID() == (*hierarchy)[top]){
37  // Find child's coordinate in dimension grouping.
38  int coord[nGroups];
39  memset(coord, 0, sizeof(int)*nGroups);
40  for(int pos=childPos, g=nGroups-1, base; g >= 0; g--) {
41  base = pow(K, g);
42  if(base <= pos) {
43  coord[g] = pos / base; // Simplified coordinate in group g.
44  pos %= base;
45  }
46  }
47  // Convert: child's coordinate in dimension grouping -> search space boundaries inside parent's subregion.
48  int nDim = region->getNDimensions();
49  int dimPerGroup = nDim / nGroups;
50  Region<P> subRegion(region);
51  Partition<P>* partition;
52  P delta, minimum;
53  for(int d=0, g=0; d < nDim; d++){
54  partition = (*region)[d]; // Perform the partitioning using the sequential order of dimension's ID.
55  delta = (partition->getEndPoint() - partition->getStartPoint()) / K;
56  minimum = partition->getStartPoint() + coord[g] * delta;
57  subRegion[d]->setStartPoint(minimum);
58  subRegion[d]->setEndPoint((coord[g]<K-1) ? minimum + delta : partition->getEndPoint());
59  if((d+1) % dimPerGroup == 0) g++; // Move to the next group.
60  }
61  hierarchy->pop_back(); // Remove the node that is in the top of ID's parentage.
62  // Re-partition the just found sub-region according to ID's descendancy.
63  return internalLoop(&subRegion, hierarchy, (*children)[childPos], ID);
64  }
65  }
