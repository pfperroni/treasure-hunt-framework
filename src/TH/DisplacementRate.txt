#ifndef DISPLACEMENTRATE_H_
#define DISPLACEMENTRATE_H_

#include "Solution.h"

template <class P = double, int pSize = 1, class F = double, int fSize = 1>
class DisplacementRate {
	int currTime;
	double rate;
	Solution<P, pSize, F, fSize> *bestSolution;
	Solution<P, pSize, F, fSize> **population;

public:
	DisplacementRate() {
		rate = 0;
		currTime = 0;
		bestSolution = NULL;
		population = NULL;
	}
	~DisplacementRate() {}

	void setCurrentTime(int currTime){ this->currTime = currTime; }

	void setBestSolution(Solution<P, pSize, F, fSize> *bestSolution) { *this->bestSolution = bestSolution; }

	void setPopulation(Solution<P, pSize, F, fSize> **population) { this->population = population; }

	void setRate(double rate) { this->rate = rate; }

	double getRate() { return rate; }
};

#endif /* DISPLACEMENTRATE_H_ */
